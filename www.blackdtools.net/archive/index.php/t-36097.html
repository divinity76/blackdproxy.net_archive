All 3 laguna islands with depositor and rs buyer.

20-25k/h on my 40~ paladin.



onGMpause
setFriendly
setMeleeKill rat
setMeleeKill wasp
setMeleeKill bug
setMeleeKill poison spider
setMeleeKill amazon
setMeleeKill toad
setMeleeKill tortoise
setMeleeKill crab
setMeleeKill Crimson Frog
setMeleeKill Coral Frog
setMeleeKill Orchid Frog
setMeleeKill Azure Frog
setMeleeKill Crocodile
setPriority Toad:+1
setLoot D7 0B
setLoot CF 0C
setLoot 0B 17
:hunt
move 32359,32907,7
IfTrue ($countitems:D2 1C$#number<=#10) Goto $nlineoflabel:depo2$
move 32348,32941,7
IfTrue ($countitems:D2 1C$#number<=#10) Goto $nlineoflabel:depo2$
move 32397,32965,7
IfTrue ($countitems:D2 1C$#number<=#10) Goto $nlineoflabel:depo2$
move 32405,32973,7

move 32434,32972,7

move 32440,32971,7

waitX 2
fastExiva > 01 00 65
